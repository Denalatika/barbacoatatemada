📊 Conexión Gratuita a Google Sheets (CRM)

Configura tu base de datos de clientes y pedidos en línea. Cuando un cliente ordene en el sitio, sus datos (Nombre, Teléfono, Tipo de Entrega, Dirección, Método de Pago, Pedido y Notas) **se escribirán automáticamente en una hoja de Google Excel en tu cuenta de Drive**. ¡Totalmente gratis y sin costos fijos!

🟢 ¡Integración activa! Los pedidos de tus clientes se enviarán de fondo a tu hoja de Google Sheets.

📘 Guía de Instalación Paso a Paso (Solo toma 3 minutos)

1 Crear la hoja de cálculo

Crea una hoja de cálculo nueva en tu Google Drive con el nombre "Pedidos Barbacoa". Cambia el nombre de la primera pestaña a "Pedidos".

2 Abrir el editor de Scripts

En la barra superior de tu hoja de cálculo, haz click en Extensiones y luego en Apps Script. Esto abrirá una pestaña de desarrollo de Google.

3 Copiar y Pegar el Código

Borra todo el código que aparezca en el archivo Código.gs de Apps Script y pega el siguiente bloque completo de código:

/*
 * Google Apps Script - Registro de Pedidos Barbacoa Tatemada
 * Versión 1.2 - Ultra-Robusta (Pestañas "Pedidos" y "Clientes" autogestionadas)
 */

// CONFIGURACIÓN DE SEGURIDAD (Opcional):
// Si creaste este script directamente desde Google Drive (y no desde Extensiones > Apps Script de tu hoja),
// pega el ID de tu hoja de cálculo entre las comillas abajo. Lo encuentras en la URL de tu hoja.
// Ejemplo: "1aBCDeFghIJKlmNoPQRStuvwXYz1234567890"
var SPREADSHEET_ID = ""; 

function doPost(e) {
  var ss;
  
  // 1. Intentar abrir la hoja de cálculo activa (para scripts contenedores)
  try {
    ss = SpreadsheetApp.getActiveSpreadsheet();
  } catch(err) {
    ss = null;
  }
  
  // 2. Si no se puede abrir como activa, abrir usando el ID proporcionado (para scripts independientes)
  if (!ss && SPREADSHEET_ID && SPREADSHEET_ID.trim() !== "") {
    try {
      ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    } catch(err) {
      ss = null;
    }
  }
  
  // 3. Si sigue sin resolverse, retornar un error claro
  if (!ss) {
    return ContentService.createTextOutput(JSON.stringify({
      "status": "error",
      "message": "Error de Configuración: No se pudo enlazar ninguna hoja de cálculo. Si creaste el script desde Drive (y no desde la hoja), asegúrate de colocar el SPREADSHEET_ID en la línea 9 de este script."
    })).setMimeType(ContentService.MimeType.JSON);
  }

  // ==========================================
  // 1. REGISTRO EN LA PESTAÑA "Pedidos"
  // ==========================================
  var sheet = ss.getSheetByName("Pedidos");
  if (!sheet) {
    sheet = ss.insertSheet("Pedidos");
  }
  
  // Crear encabezados si la hoja está totalmente vacía
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Fecha/Hora", 
      "Nombre Cliente", 
      "Teléfono", 
      "Fecha Nacimiento", 
      "Tipo de Entrega", 
      "Dirección", 
      "Ubicación / Link", 
      "Método de Pago", 
      "Estado de Pago", 
      "Monto Total", 
      "Pedido", 
      "Notas"
    ]);
    sheet.getRange("A1:L1").setFontWeight("bold").setBackground("#FFF4E0").setFontColor("#B85C2C");
  }
  
  try {
    // Validar datos de entrada
    if (!e || !e.postData || !e.postData.contents) {
      throw new Error("No se recibieron datos en el cuerpo de la petición POST.");
    }
    
    var data = JSON.parse(e.postData.contents);
    var nowText = new Date().toLocaleString("es-MX", {timeZone: "America/Hermosillo"});
    
    // Escribir fila con información detallada
    sheet.appendRow([
      nowText,
      data.customer_name || "N/A",
      "'" + (data.customer_phone || "N/A"), // Comilla para conservar ceros iniciales en Excel/Sheets
      data.customer_dob || "N/A",
      data.delivery_type || "N/A",
      data.address || "N/A",
      data.location_link || "N/A",
      data.payment_method || "N/A",
      data.transfer_status || "N/A",
      data.order_total || "N/A",
      data.order_items || "N/A",
      data.notes || "N/A"
    ]);
    
    // Auto-ajustar el ancho de las columnas de Pedidos
    sheet.autoResizeColumns(1, 12);

    // ==========================================
    // 2. REGISTRO EN LA PESTAÑA "Clientes"
    // ==========================================
    var customerSheet = ss.getSheetByName("Clientes");
    if (!customerSheet) {
      customerSheet = ss.insertSheet("Clientes");
    }

    // Crear encabezados si la hoja está vacía
    if (customerSheet.getLastRow() === 0) {
      customerSheet.appendRow([
        "Nombre", 
        "Teléfono", 
        "Domicilio", 
        "Fecha Nacimiento", 
        "Compras Realizadas", 
        "Última Compra"
      ]);
      customerSheet.getRange("A1:F1").setFontWeight("bold").setBackground("#FFEBE0").setFontColor("#B85C2C");
    }

    var customerName = data.customer_name ? data.customer_name.trim() : "N/A";
    var customerPhone = data.customer_phone ? data.customer_phone.trim() : "";
    var customerAddress = (data.delivery_type === "A domicilio" && data.address) ? data.address.trim() : "N/A";
    var customerDob = data.customer_dob ? data.customer_dob.trim() : "No proporcionada";

    if (customerPhone !== "") {
      var lastRow = customerSheet.getLastRow();
      var foundIndex = -1;
      
      // Buscar cliente por su número de teléfono de forma limpia (ignorando caracteres no numéricos)
      if (lastRow > 1) {
        var phoneRange = customerSheet.getRange(2, 2, lastRow - 1, 1);
        var phones = phoneRange.getValues();
        var cleanInputPhone = customerPhone.replace(/[^0-9]/g, "");
        
        for (var i = 0; i < phones.length; i++) {
          var cleanPhone = phones[i][0].toString().replace(/[^0-9]/g, "");
          if (cleanPhone === cleanInputPhone && cleanInputPhone !== "") {
            foundIndex = i + 2; // Arreglo empieza en 0 y la hoja en fila 2
            break;
          }
        }
      }

      if (foundIndex !== -1) {
        // Cliente existente: actualizar datos e incrementar contador
        if (customerName !== "N/A") {
          customerSheet.getRange(foundIndex, 1).setValue(customerName);
        }
        
        // Actualizar domicilio solo si esta orden fue a domicilio
        if (customerAddress !== "N/A" && customerAddress !== "") {
          customerSheet.getRange(foundIndex, 3).setValue(customerAddress);
        }
        
        // Actualizar fecha de nacimiento si no la teníamos registrada
        var currentDob = customerSheet.getRange(foundIndex, 4).getValue();
        if ((currentDob === "No proporcionada" || currentDob === "" || currentDob === "N/A") && customerDob !== "No proporcionada") {
          customerSheet.getRange(foundIndex, 4).setValue(customerDob);
        }
        
        // Incrementar el contador de compras
        var currentPurchases = parseInt(customerSheet.getRange(foundIndex, 5).getValue()) || 0;
        customerSheet.getRange(foundIndex, 5).setValue(currentPurchases + 1);
        
        // Registrar fecha y hora de la última compra
        customerSheet.getRange(foundIndex, 6).setValue(nowText);
      } else {
        // Cliente nuevo: agregar fila con contador = 1
        customerSheet.appendRow([
          customerName,
          "'" + customerPhone,
          customerAddress,
          customerDob,
          1,
          nowText
        ]);
      }
      
      // Auto-ajustar el ancho de las columnas de Clientes
      customerSheet.autoResizeColumns(1, 6);
    }
    
    return ContentService.createTextOutput(JSON.stringify({
      "status": "success",
      "message": "Pedido y datos de cliente registrados correctamente en Google Sheets."
    })).setMimeType(ContentService.MimeType.JSON);
    
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({
      "status": "error",
      "message": error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

// Habilitar GET para pruebas rápidas en navegador
function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({
    "status": "online",
    "message": "El Apps Script de Barbacoa Tatemada El Vale está funcionando correctamente y listo para recibir peticiones POST."
  })).setMimeType(ContentService.MimeType.JSON);
}

4 Implementar como Aplicación Web

1. Haz click en el botón Implementar (Deploy) en la esquina superior derecha, luego selecciona Nueva implementación (New Deployment).
2. Haz click en el engranaje de configuración y elige Aplicación Web (Web App).
3. En el campo "Ejecutar como" selecciona: Tú (tu correo electrónico).
4. En el campo "Quién tiene acceso" selecciona obligatoriamente: Cualquiera (Anyone). (Si dejas "Solo yo", las peticiones de tus clientes serán bloqueadas y el Excel no se actualizará).
5. Presiona Implementar. Google te pedirá autorizar permisos (haz click en "Autorizar acceso", luego en tu cuenta, luego en el enlace pequeño "Configuración Avanzada / Advanced" en letras grises y finalmente en "Ir a Proyecto sin nombre (no seguro)" para conceder los accesos).

⚠️ NOTA CRÍTICA DE ACTUALIZACIONES: Cada vez que realices una edición en tu código de Apps Script (por ejemplo, al añadir campos o modificar la base de datos), debes hacer una Nueva Implementación (Deploy > New Deployment) y copiar/guardar la nueva URL generada. Si solo guardas el script o no creas una nueva versión, Google seguirá ejecutando el código viejo y las actualizaciones no se reflejarán.

5 Guardar la URL en tu sitio

Al finalizar la implementación, Google te dará una URL de la Aplicación Web. Cópiala por completo, regresa a esta pestaña y pégala en el campo gris de arriba de esta página. ¡Haz click en "Guardar" y listo!
